## Installation

```bash
pnpm dreamy add toast toast-provider
```

### Add ToastProvider

You need to wrap your app with `ToastProvider` to use the toasts.

```tsx {4,6}
import { ToastProvider } from "@ui/toast-provider";

export function App() {
	return (
		<Providers>
			<ToastProvider>
				<YourApp />
			</ToastProvider>
		</Providers>
	);
}
```

## Usage

<Wrapper>
	<Button onClick={() => toast({ title: "Welcome!", description: "Make yourself at home!" })}>
		<>Toast</>
	</Button>
</Wrapper>

```tsx
function Toast() {
	const { toast } = useToast();

	return (
		<Button onClick={() => toast({ title: "Welcome!", description: "Make yourself at home!" })}>
			Toast
		</Button>
	);
}
```

### Update toast

You can use `updateToast` function to update a toast

<Wrapper>
  <UpdateToast />
</Wrapper>

```tsx
export function UpdateToast() {
	const { toast, updateToast } = useToast();
	const [toastId, setToastId] = useState<string | null>(null);

	return (
		<HStack>
			<Button
				onClick={() => {
					setToastId(
						toast({
							title: "Loading",
							description: "Please wait till file is uploaded!",
							status: "loading",
							duration: Number.POSITIVE_INFINITY
						})
					);
				}}
			>
				Send Toast
			</Button>
			<Button
				onClick={() => {
					if (toastId) {
						updateToast(toastId, {
							title: "Success!",
							description: "File uploaded successfully!",
							status: "success"
						});
					}
				}}
			>
				Update Toast
			</Button>
		</HStack>
	);
}
```

### Default toast props

Use `defaultToastProps` in the `DreamyProvider` to set default props for all toasts.

```tsx
<DreamyProvider 
  defaultToastProps={{ 
    position: "top-right",
    duration: 10_000,
    isClosable: true
  }}
>
...
</DreamyProvider>
```
