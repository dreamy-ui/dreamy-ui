# Using Dreamy UI with Next.js

## Quick Start (Recommended)

Use the automated CLI to set up Dreamy UI in your project:

```bash
npx @dreamy-ui/cli init
```

---

The CLI will automatically:

1. Detect your framework (React Router v7, Next.js, or Vite)
2. Install required dependencies:
   - Dev: `@pandacss/dev`, `@pandacss/postcss`, `@dreamy-ui/cli`
   - Prod: `@dreamy-ui/react`, `@dreamy-ui/panda-preset`, `motion`
3. Create and configure `panda.config.ts` with the Dreamy preset, patterns, and recipes
4. Add a `prepare` script (`panda codegen`) to `package.json`
5. Configure PostCSS (`postcss.config.mjs`)
6. Set up `globals.css` with Panda layers and import it in `layout.tsx`
7. Update `tsconfig.json` â€” add `styled-system/**/*` to `include`, `@/ui` and `styled-system/*` path aliases
8. Create a `components/dreamy-provider.tsx` wrapper component
9. Wire up `DreamyProvider` in `app/layout.tsx` with SSR color mode
10. Add recommended starter components (`button`, `flex`, `text`, `heading`)
11. Run Panda CSS codegen to generate `styled-system`

## Manual Installation

### 1. Install dependencies

```bash
pnpm add -D @pandacss/dev @pandacss/postcss @dreamy-ui/cli
pnpm add @dreamy-ui/react @dreamy-ui/panda-preset motion
```

### 2. Configure PostCSS

Create `postcss.config.mjs` in your project root:

```js
export default {
	plugins: {
		"@pandacss/postcss": {}
	}
};
```

### 3. Configure `panda.config.ts`

```ts {1-3, 7-16}
import createDreamyPreset, { dreamyPlugin } from "@dreamy-ui/panda-preset";
import { patterns } from "./components/patterns";
import { recipes } from "./components/recipes";

export default defineConfig({
	// ...
	jsxFactory: "dreamy",
	include: [
		"./src/**/*.{js,jsx,ts,tsx}" // Use "./app/**/*.{js,jsx,ts,tsx}" if not using a src directory
	],
	presets: [createDreamyPreset()],
	plugins: [dreamyPlugin()],
	patterns,
	theme: {
		extend: {
			recipes
		}
	}
});
```

Add components first with `npx dreamy add button` so the `components/patterns` and `components/recipes` index files exist.

### 4. Set up CSS layers

Replace the contents of `src/app/globals.css` (or `app/globals.css`) with:

```css
@layer reset, base, tokens, recipes, utilities;
```

Make sure `globals.css` is imported in your root `layout.tsx`.

### 5. Add a `prepare` script

```json
{
	"scripts": {
		"prepare": "panda codegen"
	}
}
```

### 6. Update `tsconfig.json`

```json {3-5, 10}
{
	"compilerOptions": {
		"paths": {
			"@/ui": ["./components/ui"],
			"styled-system/*": ["./styled-system/*"]
		}
	},
	"include": [
		// ...
		"styled-system/**/*"
	]
}
```

Next.js requires both the `styled-system/*` path alias and the `include` entry for proper type resolution.

### 7. Add `DreamyProvider`

Create `components/dreamy-provider.tsx` (or `src/components/dreamy-provider.tsx` if using a `src` directory):

```tsx
"use client";

import { DreamyProvider as BaseDreamyProvider } from "@dreamy-ui/react";
import type { ColorMode } from "@dreamy-ui/react";

const domMax = () => import("motion/react").then((mod) => mod.domMax);

interface DreamyProviderProps {
	children: React.ReactNode;
	colorMode?: ColorMode;
}

export function DreamyProvider({ children, colorMode }: DreamyProviderProps) {
	return (
		<BaseDreamyProvider
			motionFeatures={domMax}
			motionStrict
			colorMode={colorMode}
			useUserPreferenceColorMode
		>
			{children}
		</BaseDreamyProvider>
	);
}
```

Wire it up in `app/layout.tsx` with SSR color mode to prevent flashing:

```tsx {1-3, 6, 9, 11}
import { DreamyProvider } from "../components/dreamy-provider"; // Adjust relative path if using a src directory
import { getColorModeHTMLProps, getSSRColorMode } from "@dreamy-ui/react/rsc";
import { cookies } from "next/headers";

export default async function RootLayout({ children }) {
	const colorMode = getSSRColorMode((await cookies()).toString());

	return (
		<html lang="en" {...getColorModeHTMLProps(colorMode)}>
			<body>
				<DreamyProvider colorMode={colorMode}>{children}</DreamyProvider>
			</body>
		</html>
	);
}
```

### 8. Add components and run codegen

```bash
npx dreamy add button flex text heading
npx panda codegen
```

### 9. Start using Dreamy UI

```tsx {1, 5}
import { Button } from "@/ui";

export default function Home() {
	return <Button>Hello World</Button>;
}
```
